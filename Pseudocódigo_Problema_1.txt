Program mostrarInformacionPersonal
Start
// Programa para mostrar el nombre, la dirección, teléfono y carrera de una persona
// Declaración de variables
Declare String name
Declare String address
Declare String phone
Declare String career

// Obtener datos
Display "Por favor, ingrese su nombre: "
Input name

Input phone

Input career

// Mostrar información
Display "Información personal de ", name, ". Dirección: ", address, ". Teléfono: ", phone, ". Carrera: ", career
